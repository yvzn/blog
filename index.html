Custom React Hook for Dark Mode

A reusable React hook that handles dark mode toggle with local storage persistence. 

const useDarkMode = () => {
const [isDark, setIsDark] = useState(
localStorage.getItem('darkMode') === 'true'
);

useEffect(() => {
localStorage.setItem('darkMode', isDark);
document.body.classList.toggle('dark', isDark);
}, [isDark]);

return [isDark, setIsDark];
};

TypeScript Generic Function

A TypeScript function using generics for type-safe data handling. 

function filterArray<T>(
array: T[],
predicate: (item: T) => boolean
): T[] {
return array.filter(predicate);
}

const numbers = [1, 2, 3, 4, 5];
const evenNumbers = filterArray(
numbers,
(n) => n % 2 === 0
);

React Context Provider

A React context provider for managing global state. 

const ThemeContext = createContext();

export function ThemeProvider({ children }) {
const [theme, setTheme] = useState('light');

return (
<ThemeContext.Provider value= theme, setTheme >
	{children}
</ThemeContext.Provider>
);
}

.NET LINQ Query

A LINQ query for filtering and grouping data. 

var result = await dbContext.Orders
.Where(o => o.Status == OrderStatus.Pending)
.GroupBy(o => o.CustomerId)
.Select(g => new {
	CustomerId = g.Key,
	OrderCount = g.Count(),
	TotalAmount = g.Sum(o => o.Amount)
})
